25void JNICALL naOnAppStateChange(JNIEnv *pEnv, jobject pObj, jint state)



26{



27    /********



28     *  Android Activity life cycle states are mapped as below:



29     *  onCreate  : 1



30     *  onStart   : 2



31     *  onResume  : 3



32     *  onPause   : 4



33     *  onStop    : 5



34     *  onDestroy : 6



35     *******/



36    switch(state) {



37        case 1:



38            /*ONCREATE CALLS GOES HERE*/



39            break;



40        case 2:



41            /*ONSTART CALLS GOES HERE*/



42            break;



43        case 3:



44            onAudioCaptureStateChange(3);



45            break;



46        case 4:



47            onAudioCaptureStateChange(4);



48            break;



49        case 5:



50            /*ONSTOP CALLS GOES HERE*/



51            break;



52        case 6:



53            terminateTask(NULL);    



54            break;



55        default:



56            break;



57    }



58}

72void MW_sem_wait(sem_t *sem)



73{



74    int status;



75    while (((status = sem_wait(sem)) == -1) && (errno == EINTR)) {



76        /* Restart if interrupted by a signal */



77        continue;



78    }



79    



80    CHECK_STATUS(status, 0, "sem_wait");



81}

98void MW_blockSignal(int sigNo, sigset_t *sigMask)



99{



100    int ret;



101    sigaddset(sigMask, sigNo);



102    ret = pthread_sigmask(SIG_BLOCK, sigMask, NULL);



103    CHECK_STATUS(ret, 0, "pthread_sigmask");



104}

106void MW_setTaskPeriod(double periodInSeconds, int sigNo, sighandler_t handler)



107{



108    timer_t timerId;



109    struct sigevent sev;



110    struct itimerspec its;



111    long stNanoSec;



112    int ret;



113    



114    /* before creating timer lets initialize the signal handler */



115    signal(sigNo, handler);



116    



117    /* Create a timer */



118    sev.sigev_notify = SIGEV_SIGNAL;



119    sev.sigev_signo = sigNo;



120    sev.sigev_value.sival_ptr = &timerId;



121    ret = timer_create(CLOCK_REALTIME, &sev, &timerId);



122    CHECK_STATUS(ret, 0, "timer_create");



123    



124    /* Arm real-time scheduling timer */



125    stNanoSec = (long)(periodInSeconds * 1e9);



126    its.it_value.tv_sec = stNanoSec / 1000000000;



127    its.it_value.tv_nsec = stNanoSec % 1000000000;



128    its.it_interval.tv_sec = its.it_value.tv_sec;



129    its.it_interval.tv_nsec = its.it_value.tv_nsec;



130    ret = timer_settime(timerId, 0, &its, NULL);



131    CHECK_STATUS(ret, 0, "timer_settime");



132}

134void flashMessage(const char* inMessage)



135{



136    if (flashMessageID != NULL) {



137        JNIEnv *pEnv;



138        jstring jMessage;



139        (*cachedJvm)->AttachCurrentThread(cachedJvm, &pEnv, NULL);



140        jMessage = (*pEnv)->NewStringUTF(pEnv, inMessage);



141        (*pEnv)->CallVoidMethod(pEnv, cachedActivityObj, flashMessageID, jMessage);



142        (*pEnv)->DeleteLocalRef(pEnv, jMessage);



143    }



144}

165JNIEXPORT jint JNICALL JNI_OnLoad(JavaVM* pVm, void *reserved)



166{



167    JNIEnv* env;



168    if ((*pVm)->GetEnv(pVm, (void **)&env, JNI_VERSION_1_6) != JNI_OK) {



169        LOGMESSAGE_ARG1("Not version %d; Exiting", JNI_VERSION_1_6);



170        return -1;



171    }



172    



173    JNINativeMethod nm[2];



174    nm[0].name = "naMain";



175    nm[0].signature =



176            "([Ljava/lang/String;Lcom/example/matlabfcn_whistlecount/matlabfcn_whistlecount;)I";



177    nm[0].fnPtr = (void*)naMain;



178    nm[1].name = "naOnAppStateChange";



179    nm[1].signature =



180            "(I)V";



181    nm[1].fnPtr = (void*)naOnAppStateChange;



182    



183    jclass cls = (*env)->FindClass(env,



184            "com/example/matlabfcn_whistlecount/matlabfcn_whistlecount");



185    



186    //Register methods with env->RegisterNatives.



187    (*env)->RegisterNatives(env, cls, nm, 2);



188    return JNI_VERSION_1_6;



189}

205void myRTOSInit(double baseRatePeriod, int numSubrates)



206{



207    int ret;



208    pthread_attr_t attr;



209    pthread_t baseRateThread;



210    size_t stackSize;



211    unsigned long cpuMask = 0x1;



212    unsigned int len = sizeof(cpuMask);



213    



214    /* Flash a message in the main activity */



215    flashMessage("Starting the model");



216    



217    /* Initialize semaphore used for thread synchronization */



218    ret = sem_init(&stopSem, 0, 0);



219    CHECK_STATUS(ret, 0, "sem_init:stopSem");



220    



221    /* Create threads executing the Simulink model */



222    ret = pthread_attr_init(&attr);



223    CHECK_STATUS(ret, 0, "pthread_attr_init");



224    ret = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);



225    CHECK_STATUS(ret, 0, "pthread_attr_setdetachstate");



226    



227    /* Block signal used for timer notification */



228    baseRateInfo.period = baseRatePeriod;



229    baseRateInfo.sigNo = SIGRTMIN;



230    sigemptyset(&baseRateInfo.sigMask);



231    MW_blockSignal(baseRateInfo.sigNo, &baseRateInfo.sigMask);



232    ret = pthread_mutex_init (&baseRateInfo.sigWaitMutex, NULL);



233    CHECK_STATUS(ret, 0, "pthread_mutex_init");



234    ret = pthread_cond_init (&baseRateInfo.sigWait, NULL);



235    CHECK_STATUS(ret, 0, "pthread_cond_init");



236    signal(SIGTERM, MW_exitHandler);     /* kill */



237    signal(SIGHUP, MW_exitHandler);      /* kill -HUP */



238    signal(SIGINT, MW_exitHandler);      /* Interrupt from keyboard */



239    signal(SIGQUIT, MW_exitHandler);     /* Quit from keyboard */



240    



241    MW_setTaskPeriod(baseRateInfo.period, baseRateInfo.sigNo, MW_baseRateSignalHandler);



242    



243    /* Create base rate task */



244    ret = pthread_create(&baseRateThread, &attr, (void *) baseRateTask, (void *)



245    &baseRateInfo);



246    CHECK_STATUS(ret, 0, "pthread_create");



247    



248    ret = pthread_attr_destroy(&attr);



249    CHECK_STATUS(ret, 0, "pthread_attr_destroy");



250    



251}
